def remove_outliers(col):
    sorted(col)
    Q3,Q1 = np.percentile(col,[75,25])
    IQR = Q3-Q1
    uplimit = (1.5*IQR)+Q3
    lwlimit = (1.5*IQR)-Q1
    return uplimit,lwlimit
